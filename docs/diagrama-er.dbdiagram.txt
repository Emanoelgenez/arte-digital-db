// Plataforma de Análise e Produção de Arte Digital

Table Artista {
  id_artista int [pk, increment]
  nome varchar(100)
  bio text
  pais_origem varchar(50)
}

Table Estilo {
  id_estilo int [pk, increment]
  nome varchar(100)
  descricao text
}

Table Categoria {
  id_categoria int [pk, increment]
  nome varchar(100)
  descricao text
}

Table Obra {
  id_obra int [pk, increment]
  titulo varchar(150)
  descricao text
  data_criacao date
  id_artista int [ref: > Artista.id_artista]
  id_estilo int [ref: > Estilo.id_estilo]
  id_categoria int [ref: > Categoria.id_categoria]
}

Table Ferramenta {
  id_ferramenta int [pk, increment]
  nome varchar(100)
  tipo varchar(50)
}

Table Obra_Ferramenta {
  id_obra int [ref: > Obra.id_obra]
  id_ferramenta int [ref: > Ferramenta.id_ferramenta]
  Note: "Relacionamento N:N entre obras e ferramentas"
  indexes {
    (id_obra, id_ferramenta) [pk]
  }
}

Table Feedback {
  id_feedback int [pk, increment]
  id_obra int [ref: > Obra.id_obra]
  usuario varchar(100)
  comentario text
  nota_composicao int
  nota_cor int
  nota_mensagem int
}
